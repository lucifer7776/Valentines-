<script>
const pageAsk = document.getElementById("pageAsk");
const pageDate = document.getElementById("pageDate");

const yesBtn = document.getElementById("yesBtn");
const noBtn = document.getElementById("noBtn");
const btnArea = document.getElementById("btnArea");

const playBtn = document.getElementById("playBtn");
const pauseBtn = document.getElementById("pauseBtn");
const music = document.getElementById("music");

const hearts = document.getElementById("hearts");
const confetti = document.getElementById("confetti");

const dateGrid = document.getElementById("dateGrid");
const finalBox = document.getElementById("finalBox");
const finalLine = document.getElementById("finalLine");
const backBtn = document.getElementById("backBtn");

const typeLine = document.getElementById("typeLine");

const heartChars = ["üíó","üíñ","üíò","üíï","‚ù§Ô∏è","üåπ"];
function spawnHeart(){
  const s = document.createElement("span");
  s.textContent = heartChars[Math.floor(Math.random()*heartChars.length)];
  const left = Math.random()*100;
  const size = 14 + Math.random()*28;
  const dur = 4 + Math.random()*4.5;
  const drift = (Math.random() * 120 - 60) + "px";
  s.style.left = left + "vw";
  s.style.fontSize = size + "px";
  s.style.animationDuration = dur + "s";
  s.style.setProperty("--drift", drift);
  hearts.appendChild(s);
  setTimeout(() => s.remove(), (dur + 0.2) * 1000);
}
setInterval(spawnHeart, 220);

function launchConfetti(){
  confetti.innerHTML = "";
  confetti.style.display = "block";
  const pieces = 160;
  for(let i=0;i<pieces;i++){
    const c = document.createElement("i");
    const left = Math.random()*100;
    const dur = 2.2 + Math.random()*2.4;
    const delay = Math.random()*0.15;
    const hue = 330 + Math.floor(Math.random()*40);
    c.style.left = left + "vw";
    c.style.background = `hsl(${hue}, 95%, 60%)`;
    c.style.animationDuration = dur + "s";
    c.style.animationDelay = delay + "s";
    confetti.appendChild(c);
  }
  setTimeout(()=> confetti.style.display = "none", 4200);
}

function typewriter(el, text, speed=34){
  el.innerHTML = "";
  const span = document.createElement("span");
  span.textContent = "";
  const cur = document.createElement("span");
  cur.className = "cursor";
  cur.textContent = "|";
  el.appendChild(span);
  el.appendChild(cur);

  let i = 0;
  const tick = () => {
    span.textContent = text.slice(0, i);
    i++;
    if(i <= text.length){
      setTimeout(tick, speed);
    }
  };
  tick();
}

typewriter(typeLine, "Bear is asking you officially üòåüíò");

async function tryPlay(){
  try{
    await music.play();
    playBtn.style.display = "none";
    pauseBtn.style.display = "inline-block";
  }catch(e){}
}

playBtn.addEventListener("click", tryPlay);

pauseBtn.addEventListener("click", () => {
  music.pause();
  pauseBtn.style.display = "none";
  playBtn.style.display = "inline-block";
});

yesBtn.addEventListener("click", async () => {
  await tryPlay();
  launchConfetti();

  setTimeout(() => {
    pageAsk.classList.remove("active");
    pageDate.classList.add("active");
    launchConfetti();
  }, 700);
});

function moveNoButton(){
  const areaRect = btnArea.getBoundingClientRect();
  const btnRect = noBtn.getBoundingClientRect();
  const maxX = areaRect.width - btnRect.width;
  const maxY = areaRect.height - btnRect.height;
  const x = Math.max(0, Math.floor(Math.random() * maxX));
  const y = Math.max(0, Math.floor(Math.random() * maxY));
  noBtn.style.left = x + "px";
  noBtn.style.top = y + "px";
}
noBtn.addEventListener("mouseenter", moveNoButton);
noBtn.addEventListener("touchstart", (e) => {
  e.preventDefault();
  moveNoButton();
}, {passive:false});
noBtn.addEventListener("click", moveNoButton);

dateGrid.addEventListener("click", (e) => {
  const opt = e.target.closest(".opt");
  if(!opt) return;
  const choice = opt.getAttribute("data-choice");
  finalLine.textContent = `Janae picked: ${choice}. Bear is planning it üíó`;
  finalBox.style.display = "block";
  launchConfetti();
});

backBtn.addEventListener("click", () => {
  pageDate.classList.remove("active");
  pageAsk.classList.add("active");
  finalBox.style.display = "none";
});
